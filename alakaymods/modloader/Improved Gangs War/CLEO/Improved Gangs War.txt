    0AF1: write_int 0@ to_ini_file "cleo\Improved Gangs War.ini" section "Models" key "gngblproof2"

    if
        player.Defined(0)
    then
        if
            087A:   gang_war_in_progress
        then
        
            0A8D: 2@  = read_memory 0x96AB64 size 1 virtual_protect 0 //  mov     eax, ds:_ZN9CGangWars5StateE ; CGangWars::State

                Scr_visual == 1

                continue

                Scr_bulletproof == 1

                continue

                Scr_faster == 1
                1@ == 0
            then
                0A92: stream_custom_script "Faster.qa" Speed_anim Improved_Gangs_Spawn
                wait 1000
                continue
            end
            
            
        end
